#include <iostream>
using namespace std;

class Student {
private:
    string name;
    int rollNumber;

public:
    Student(string n, int r) {
        name = n;
        rollNumber = r;
    }

    void display() {
        cout << "Name: " << name << ", Roll No: " << rollNumber << endl;
    }
};

int main() {
    int n;
    cout << "Enter number of students: ";
    cin >> n;

    Student* students[n];

    for (int i = 0; i < n; i++) {
        string name;
        int roll;
        cout << "Enter name and roll number of student " << i + 1 << ": ";
        cin >> name >> roll;
        students[i] = new Student(name, roll);
    }

    cout << "\nStudent Details:\n";
    for (int i = 0; i < n; i++) {
        students[i]->display();
    }

    return 0;
}
